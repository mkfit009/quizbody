<script>
function checkAnswer(event, quizId, correctAnswer) {
  event.preventDefault();
  const form = event.target;
  const selected = form.querySelector('input[type="radio"]:checked');
  const resultDiv = document.getElementById('result-' + quizId);
  
  if (!selected) {
    resultDiv.textContent = "Seleziona una risposta.";
    resultDiv.style.color = "red";
    return;
  }

  if (selected.value === correctAnswer) {
    resultDiv.textContent = "Corretto!";
    resultDiv.style.color = "green";
  } else {
    resultDiv.textContent = "Risposta errata.";
    resultDiv.style.color = "red";
  }
}
</script>

</body>
</html>
